\subsection{Arithmetische Mittel}
\begin{equation}
\overline{x} = \frac{1}{n}(x_1+x_2+...+x_n)=\frac{1}{n}\sum_{i=1}^{n}x_i
\end{equation}
\begin{center}\begin{tabular}{rcl}
   $x_i$ & = & i-te Merkmalsausprägung  \\
   $n$ & = & Anzahl der Merkmalsausprägungen \\
\end{tabular}\end{center}
